# Step 1 — Make sure you're on main
git checkout main

# Step 2 — Verify your files are committed
git status
git add .
git commit -m "your message"

# Step 3 — Connect to GitHub (if not already)
git remote add origin https://github.com/username/repo-name.git

# Step 4 — Push main → another branch on GitHub
git push origin main:your-branch-name
